 23    >>> help(sg)

 28    >>> pymeasure.デバイスの種類(コマンドの種類, 通信方法, [通信方法の設定])

 71 -def create_communicator(communication_method, *args, **kwargs): 

 72      """ 
 73      通信インスタンスを生成します。 
 74   
 75      Parameters 
 76      ---------- 
 77      communication_method : 文字列 
 78          'Ethernet', 'LAN' --> Ethernet通信インスタンスを返します。 
 79              必要な引数 : host, port 
 80          'GPIB_Prologix', 'Prologix' --> GPIB_Prologix通信インスタンスを返します。 
 81              必要な引数 : host, port, gpib 
 82          'NI_USB' --> NI_DAQ用のUSB通信インスタンスを返します。 
 83              必要な引数 : devname 
 84   
 85          文字列の大文字小文字は区別しません。 
 86   
 87      Returns 
 88      ------- 
 89      com : 通信インスタンス 
 90   
 91      Notes 
 92      ----- 
 93   
 94      Examples 
 95      -------- 
 96      >>> com = pymeasure.create_communicator('Ethernet', 
 97                                              host='192.168.100.11', port=5025) 
 98      """ 
 99      method = communication_method.lower() 
100   
101      if method in ['ethernet', 'lan']: 
102          return communicators.ethernet(*args, **kwargs) 
103   
104      elif method in ['gpib_prologix', 'gpib-prologix', 'prologix']: 
105          return communicators.gpib_prologix(*args, **kwargs) 
106   
107      elif method in ['ni_usb']: 
108          return communicators.ni_usb(*args, **kwargs) 
109   
110      return None
